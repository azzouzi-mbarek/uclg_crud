import { FormGroup, FormBuilder } from '@angular/forms';
import { BsModalRef } from 'ngx-bootstrap/modal';
import { Component, OnInit } from '@angular/core';

import { Subject } from 'rxjs';

import * as insQuery from './../institutionQueries';
import { InstitutionCategory } from '../InstitutionCategory.model';
import { ErrorHandlerService } from 'src/app/modules/auth/error-handler.service';
import { InstitutionCategoryService } from '../institution-category.service';


@Component({
  selector: 'app-modal-form-institution-category',
  templateUrl: './modal-form-institution-category.component.html',
  styleUrls: ['./modal-form-institution-category.component.css']
})
export class ModalFormInstitutionCategoryComponent implements OnInit {


  // modal parms
  title: String;
  id: number;
  state: Boolean;
  SubmitBtnTitle: String;

  // category params
  category: InstitutionCategory;
  categorySubject: Subject<{}>;
  errors: any;

  // form category
  institutionCategoryForm: FormGroup;
  constructor(
    private _BsModalRef: BsModalRef,
    private _institutionCategoryService: InstitutionCategoryService,
    private _errorHandler: ErrorHandlerService,
    private _fb: FormBuilder) { }

  ngOnInit() {

    this.institutionCategoryForm = this._fb.group({
      name: ['']
    });


    // this.initErrors('name');


    if (this.category) {
      // state == true : open modal on edit mode
      this.state = this.category !== null ? true : false;
    } else {
      this.category = new InstitutionCategory();
    }
    this.categorySubject = new Subject();

    this.title = !this.state ? 'Nouvelle Catégorie :' : 'Éditer ' + this.category.name;
    this.SubmitBtnTitle = !this.state ? 'Sauvegarder' : 'Éditer';

    this.institutionCategoryForm.patchValue(this.category);
  }

  initErrors(field) {
    this.institutionCategoryForm.get(field).valueChanges.subscribe(
      value => {
        delete this.errors['name'];
        console.log(this.errors);
      }
    );
  }



  onDecline() {
    this._BsModalRef.hide();
  }

  mapFromValueToCategory() {
    this.category.name = this.institutionCategoryForm.value.name;
  }

  onSubmit() {

    this.mapFromValueToCategory();
    const mutation = this.state ? insQuery.MUTATION_UPDATE_INSTITUTION_CATEGORY : insQuery.MUTATION_CREATE_INSTITUTION_CATEGORY;

    this._institutionCategoryService.institutionCategoryMutation(mutation, this.category)
      .subscribe(
        (res: any) => {
          const data = {
            category: this.state ? res.data.UpdateInstitutionCategory : res.data.CreateInstitutionCategory,
            edit: this.state ? true : false
          };
          this.categorySubject.next(data);
          this._BsModalRef.hide();
          this.category = new InstitutionCategory();

        },
        (errors) => {
          this.errors = this._errorHandler.handleError(errors);
          console.log(this.errors);
        }
      );


  }



}
